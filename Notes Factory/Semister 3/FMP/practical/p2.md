multiplication code of 2 8bit numbers (h3)

..model small
.data

a db 04h
b db 02h

.code
mov ax,@data
mov ds,ax

mov al,a
mov bl,b
mul bl
mov ah,02h

mov ch,02h
mov cl,04h

mov bh,al
i2:rol bh,cl
mov dl,bh

and dl,0fh
cmp dl,09
jbe i4
add dl,07

i4:add dl,30h
mov ah,02h

int 21h
dec ch
jnz I2

mov ah,4ch
int 21h

end
