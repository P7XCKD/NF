8 bit addition fmp code
```
.model small

.data
a db 02h
b db 02h

.code 
mov ax,@data
mov ds,ax

mov al,a
mov bl,b
add al,bl
mov dl,al

add dl,30h
mov ah,02h
int 21h
mov ah,4ch
int 21h

end
```

output = 4

perform 8 bit addition for 13h and 02h 
```
.model small

.data
a db 13h
b db 02h

.code
mov ax,@data

mov bl,b
add al,bl

mov ch,02h
mov cl,04h
mov bh,al

I2:rol bh,cl
mov dl,bh
and dl,0fh
cmp dl,09
jbe I4

add dl,07

mov ah,02h

end
```
