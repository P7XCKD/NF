8 bit addition fmp code
```
.model small

.data
a db 02h
b db 02h

.code 
mov ax,@data
mov ds,ax

mov al,a
mov bl,b
add al,bl
mov dl,al

add dl,30h
mov ah,02h
int 21h
mov ah,4ch
int 21h

end
```

output = 4
